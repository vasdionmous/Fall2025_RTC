\begin{exercise}
    \label{ex:1.1}
    Να υπολογίσετε την ορίζουσα και το ίχνος του πίνακα 
    \[
    A = 
    \begin{pmatrix}
        1 & 0 & 0 \\
        0 & 2 & 0 \\
        0 & 0 & 3
    \end{pmatrix}.
    \]
\end{exercise}

\begin{solution}
Η ορίζουσα και το ίχνος του πίνακα είναι 

\begin{alignat*}{2}
    \det(A)   & = 1 \cdot 2 \cdot 3 && = 6 \\
    \trace(A) & = 1 + 2 + 3         && = 6.
\end{alignat*}

\end{solution}
