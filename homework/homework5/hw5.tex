Για $\lambda \vdash n$, έστω $\chi^\lambda$ ο χαρακτήρας του προτύπου Specht $\sS^\lambda$.

\begin{exercise} 
    Να δείξετε τα εξής.
    \begin{itemize}
        \item[(1)] Αν $\lambda \vdash n$ και $0 \le k \le n-1$, τότε
        \[
        \chi^\lambda((n)) = 
        \begin{cases}
            (-1)^k, &\ \text{αν $\lambda = (n-k,1^k)$} \\
            0, &\ \text{διαφορετικά}.
        \end{cases}
        \]
        \item[(2)] Για κάθε $n \ge 1$,
        \[
        p_n = \sum_{k=0}^{n-1} (-1)^k s_{(n-k,1^k)}.
        \]
    \end{itemize}
    \noindent
    \textcolor{lightgray}{\small{\emph{Υπόδειξη}: Χρησιμοποιήστε τον κανόνα Murnaghan--Nakayama.}}
\end{exercise}

\begin{exercise}{\texttt{(Εκτιμήσεις συμμετρικών συναρτήσεων)}} \\
    Ο ομομορφισμός δακτύλιων $\ps_n : \Sym \to \CC$ που ορίζεται θέτοντας 
    \[
    \ps_n(f) = f(1^n) = f(\underbrace{1,1,\dots,1}_{\text{$n$ το πλήθος}},0,0,\dots)
    \]
    για κάθε $f \in \Sym$ ονομάζεται \defn{πρωταρχική εκτίμηση} τάξης $n$. Αν $\lambda$ είναι διαμέριση ακεραίου, να υπολογίσετε το $\ps_n(f_\lambda)$, για κάθε $f \in \{e, h, p\}$.

    \noindent
    \textcolor{lightgray}{\small{\emph{Σημείωση}: Για την περίπτωση της πλήρους ομογενής συμμετρικής συνάρτησης μπορείτε να χρησιμοποιήσετε γνωστή πρόταση από τα Διακρικά Μαθηματικά.}}
\end{exercise}

\begin{exercise}
    Για $\lambda \vdash n$, να δείξετε ότι
    \begin{align}
        \tag{1} \omega(p_\lambda) &= \sign(\lambda) p_\lambda \\
        \tag{2} \sS^{\lambda^\top} &\cong_{\fS_n} \sS^\lambda \otimes \sS^{(1^n)},
    \end{align}
    όπου $\sign(\lambda) = (-1)^{n-\ell(\lambda)}$.

    \noindent
    \textcolor{lightgray}{\small{\emph{Υπόδειξη}: Για το (2) μπορείτε να χρησιμοποιήσετε το (1) και το ανάπτυγμα των power sum συμμετρικών συναρτήσεων στην βάση των συναρτήσεων Schur.}}
\end{exercise}
